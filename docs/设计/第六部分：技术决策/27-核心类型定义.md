# 核心类型定义

## Task

```rust
pub struct Task {
    pub id: u64,
    pub scene_id: u64,
    pub priority: Priority,
    pub kind: TaskKind,
    pub payload: Vec<u8>,
    pub created_at: Instant,
    pub state: TaskState,
    pub retry_count: u8,
    pub max_retries: u8,
    pub timeout: Duration,
    pub assigned_to: Option<u64>,
    pub tags: Vec<String>,
}

pub enum TaskKind { Reasoning, ToolCall, FileOp, Browser, Knowledge }

pub enum Priority { Emergency = 0, High = 1, Normal = 4, Low = 8, Background = 16 }

pub enum TaskState { Pending, Queued, Running, Completed, Failed, TimedOut, DeadLetter, Canceled }
```

## Memory

```rust
pub struct Memory {
    pub agent_id: u64,
    pub scene_id: u64,
    pub scope: MemoryScope,
    pub owners: Vec<OwnerId>,
    pub category: MemoryCategory,
    pub subagent_type: SubAgentType,
    pub enhancement_rate: EnhancementRate,
    pub decay_rate: DecayRate,
    pub strength_cached: f64,
    pub last_updated: Timestamp,
    pub last_validated: Timestamp,
    pub validation_count: u32,
    pub source_count: u32,
    pub seq: u32,
    pub embedding: Option<Vec<f32>>,  // 可选，不索引时不存储
}

pub enum MemoryScope { Single(SceneId), All, Partial(Vec<SceneId>) }

pub enum MemoryCategory { Knowledge, Habit, Experience, Rule, Relation }

pub enum OwnerId { Agent(u64), AgentType(SubAgentType), All }
```

## Agent

```rust
pub struct MainAgent {
    pub identity: AgentIdentity,
    pub rules: AgentRules,
    pub scene: SceneId,
    pub model: ModelId,
}

pub struct SubAgent {
    pub id: u64,
    pub state: SubAgentState,
    pub current_tasks: u32,
    pub max_concurrent: u32,
    pub started_at: Instant,
    pub last_heartbeat: Instant,
    pub total_completed: u64,
    pub total_failed: u64,
    pub avg_completion_time: Duration,
    pub loaded_resources: Vec<String>,
    pub recent_task_types: Vec<TaskKind>,
}

pub enum SubAgentState { Idle, Busy, Overloaded, Unhealthy, Dead }
```

## Trust

```rust
pub struct TrustSource {
    pub agent_id: u64,
    pub credibility: f64,       // 0.0 ~ 1.0
    pub last_validated: Timestamp,
    pub wrong_validations: u32,
    pub correct_validations: u32,
}

pub struct EnhancementRate(pub f64);  // +X/天
pub struct DecayRate(pub f64);        // -X/天
```

## Scene

```rust
pub struct Scene {
    pub id: u64,
    pub name: String,
    pub paths: Vec<PathBuf>,
    pub config_dir: PathBuf,
    pub memory_summary: MemorySummary,
    pub memory_full: Option<MemoryFull>,
}

pub struct SceneConfig {
    pub tools: ToolConfig,
    pub trust: TrustConfig,
    pub memory: MemoryConfig,
}

pub enum SceneDetectionRule {
    DotBugsDir,
    BugMdFile,
    GitRepo,
    Custom(PathBuf),

```

## Browser

```rust
pub struct BrowserManager {
    pub headless_pool: ContextPool,
    pub headful_pool: ContextPool,
}

pub struct ContextPool {
    pub contexts: Vec<BrowserContext>,
    pub max_size: usize,
    pub idle_timeout: Duration,
    pub max_tabs_per_context: usize,
}

pub struct BrowserContext {
    pub id: u64,
    pub tabs: Vec<Tab>,
    pub fingerprint: BrowserFingerprint,
    pub cookie_jar: CookieJar,
    pub state: ContextState,
}

pub enum ContextState { Active, Dormant, Dead }

pub struct BrowserFingerprint {
    pub user_agent: String,
    pub canvas_noise: f32,
    pub webgl_hash: String,
    pub screen_resolution: (u16, u16),
    pub fonts: Vec<String>,
    pub locale: String,
}
```
